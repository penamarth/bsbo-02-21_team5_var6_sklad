```plantuml
@startuml
actor Manager
participant "Information System Warehouse" as ISW
participant "Goods Accounting System" as GAS

autonumber

Manager -> ISW : Open catalog
activate Manager
activate ISW
ISW -> GAS : Enter product barcode
activate GAS
GAS -> ISW : Return product information
ISW -> Manager : Display product information
Manager -> ISW : Click "Ship"
ISW -> GAS : Confirm quantity for shipping
GAS -> ISW : Update product information
ISW -> Manager : Update successful
deactivate Manager
deactivate ISW
deactivate GAS

@enduml
